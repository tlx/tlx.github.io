23 /*!
24  * A ring (circular) buffer of static (non-growing) size.
25  *
26  * Due to many modulo operations with capacity_, the capacity is rounded up to
27  * the next power of two, even for powers of two! This is because otherwise
28  * size() == end - begin == 0 after filling the ring buffer, and adding another
29  * size_ member requires more book-keeping.
30  */
31 template <typename Type, class Allocator = std::allocator<Type> >
32 class RingBuffer

64  //! copy-constructor: create new ring buffer

78  //! copyable: create new ring buffer
79  RingBuffer& operator = (const RingBuffer& rb) {
80  if (this == &rb) return *this;
81  // empty this buffer
82  clear();
83  // reallocate buffer if the size changes
84  if (capacity_ != rb.capacity_ || alloc_ != rb.alloc_)
85  {
86  alloc_.deallocate(data_, capacity_);
87  alloc_ = rb.alloc_;
88  capacity_ = rb.capacity_;
89  data_ = alloc_.allocate(capacity_);
90  }
91  // copy over fields
92  max_size_ = rb.max_size_;
93  mask_ = rb.mask_;
94  begin_ = end_ = 0;
95  // copy over items (we cannot just flat copy the array due to item
96  // construction).
97  for (size_t i = 0; i < rb.size(); ++i)
98  push_back(rb[i]);
99  return *this;
100  }

323  alloc_.deallocate(data_, capacity_);
324  }
325 
326  // setup buffer
