49  : key_(key), longkey_(longkey), keytype_(keytype), desc_(desc),
50  required_(required) { }
51 
52  //! empty virtual destructor
53  virtual ~Argument() = default;

63 
64  //! return 'longkey [keytype]'
65  std::string param_text() const {
66  std::string s = longkey_;
67  if (keytype_.size()) {
68  s += ' ' + keytype_;
69  }
70  return s;
71  }
72

75  std::string s;

82  }
83  return s;

357 void CmdlineParser::output_wrap(
358  std::ostream& os, const std::string& text,
359  size_t wraplen, size_t indent_first, size_t indent_rest, size_t current,
360  size_t indent_newline) {
361 
362  std::string::size_type t = 0;
363  size_t indent = indent_first;
364 
365  while (t != text.size()) {
366  std::string::size_type to = t, lspace = t;
367 
368  // scan forward in text until we hit a newline or wrap point
369  while (to != text.size() && to + current + indent < t + wraplen &&
370  text[to] != '\n') {
371  if (text[to] == ' ')
372  lspace = to;
373  ++to;
374  }
375 
376  // go back to last space
377  if (to != text.size() && text[to] != '\n' && lspace != t)
378  to = lspace + 1;
379 
380  // output line
381  os << std::string(indent, ' ') << text.substr(t, to - t) << std::endl;
382 
383  current = 0;
384  indent = indent_rest;
385 
386  // skip over last newline
387  if (to != text.size() && text[to] == '\n') {
388  indent = indent_newline;
389  ++to;
390  }
391 
392  t = to;
393  }
394 }

650 void CmdlineParser::print_usage(std::ostream& os) {
651  std::ios::fmtflags flags(os.flags());
652 
653  os << "Usage: " << program_name_
654  << (option_list_.size() ? " [options]" : "");
655 
656  for (ArgumentList::const_iterator it = param_list_.begin();
657  it != param_list_.end(); ++it) {
658  const Argument* arg = *it;
659 
660  os << (arg->required_ ? " <" : " [") << arg->longkey_
661  << (arg->repeated_ ? " ..." : "") << (arg->required_ ? '>' : ']');
662  }
663 
664  os << std::endl;
665 
666  if (description_.size()) {

738 bool CmdlineParser::process(
739  int argc, const char* const* argv, std::ostream& os) {
740  program_name_ = argv[0];
741  --argc, ++argv;
742 
743  // search for help string and output help
744  for (int i = 0; i < argc; ++i) {
745  if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
746  print_usage(os);
747  return false;
748  }
749  }
750 
751  // current argument in param_list_
752  ArgumentList::iterator argi = param_list_.begin();
753  bool end_optlist = false;
754 
755  while (argc != 0) {
756  const char* arg = argv[0];
757 
758  if (arg[0] == '-' && !end_optlist) {
759  // option, advance to argument
760  --argc, ++argv;
761  if (arg[1] == '-') {
762  if (arg[2] == '-') {
763  end_optlist = true;
764  }
765  else {
766  // long option
767  ArgumentList::const_iterator oi = option_list_.begin();
768  for ( ; oi != option_list_.end(); ++oi) {
769  if ((arg + 2) == (*oi)->longkey_) {
770  if (!(*oi)->process(argc, argv)) {
771  print_option_error(argc, argv, *oi, os);
772  return false;
773  }
774  else if (verbose_process_) {
775  os << "Option " << (*oi)->option_text()
776  << " set to ";
777  (*oi)->print_value(os);
778  os << '.' << std::endl;
779  }
780  break;
781  }
782  }
783  if (oi == option_list_.end()) {
784  os << "Error: unknown option \"" << arg << "\"."
785  << std::endl << std::endl;
786  print_usage(os);
787  return false;
788  }
789  }
790  }
791  else {
792  // short option
793  if (arg[1] == 0) {
794  os << "Invalid option \"" << arg << "\"." << std::endl;
795  }
796  else {
797  size_t offset = 1, arg_length = strlen(arg);
798  int old_argc = argc;
799  // Arguments will increase argc, so abort if it increases,
800  // while flags won't, so increase offset and parse next
801  while (offset < arg_length && argc == old_argc) {
802  ArgumentList::const_iterator oi = option_list_.begin();
803  for ( ; oi != option_list_.end(); ++oi) {
804  if (arg[offset] == (*oi)->key_) {
805  ++offset;
806  if (!(*oi)->process(argc, argv)) {
807  print_option_error(argc, argv, *oi, os);
808  return false;
809  }
810  else if (verbose_process_) {
811  os << "Option "
812  << (*oi)->option_text()
813  << " set to ";
814  (*oi)->print_value(os);
815  os << '.' << std::endl;
816  }
817  break;
818  }
819  }
820  if (oi == option_list_.end()) {
821  os << "Error: unknown option \"";
822  if (arg_length > 2) {
823  // multiple short options combined
824  os << "-" << arg[offset]
825  << "\" at position " << offset
826  << " in option sequence \"";
827  }
828  os << arg << "\"." << std::endl << std::endl;
829  print_usage(os);
830  return false;
831  }
832  }
833  }
834  }
835  }
836  else {
837  if (argi != param_list_.end()) {
838  if (!(*argi)->process(argc, argv)) {
839  print_param_error(argc, argv, *argi, os);
840  return false;
841  }
842  else if (verbose_process_) {
843  os << "Parameter " << (*argi)->param_text() << " set to ";
844  (*argi)->print_value(os);
845  os << '.' << std::endl;
846  }
847  (*argi)->found_ = true;
848  if (!(*argi)->repeated_)
849  ++argi;
850  }
851  else {
852  os << "Error: unexpected extra argument "
853  << "\"" << argv[0] << "\"." << std::endl << std::endl;
854  --argc, ++argv;
855  print_usage(os);
856  return false;
857  }
858  }
859  }
860 
861  bool good = true;
862 
863  for (ArgumentList::const_iterator it = param_list_.begin();
864  it != param_list_.end(); ++it) {
865  if ((*it)->required_ && !(*it)->found_) {
866  os << "Error: argument for parameter " << (*it)->longkey_
867  << " is required!" << std::endl;
868  good = false;
869  }
870  }
871 
872  if (!good) {
873  os << std::endl;
874  print_usage(os);
875  }
876 
877  return good;
878 }
