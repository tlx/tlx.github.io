23 /*!
24  * Simpler non-growing vector without initialization.
25  *
26  * SimpleVector can be used a replacement for std::vector when only a
27  * non-growing array of simple types is needed. The advantages of SimpleVector
28  * are that it does not initilize memory for POD types (-> faster), while normal
29  * structs are supported as well if default-contractible. The simple pointer
30  * types allow faster compilation and is less error prone to copying and other
31  * problems.
32  */
33 template <typename ValueType>
34 class SimpleVector
35 {
36 public:
37  using value_type = ValueType;
38  using size_type = size_t;
39 
40 protected:
41  //! size of allocated memory
42  size_type size_;
43 
44  //! pointer to allocated memory area
45  value_type* array_;
46 
47 public:
48  // *** simple pointer iterators
49 
50  using iterator = value_type *;
51  using const_iterator = const value_type *;
52  using reference = value_type &;
53  using const_reference = const value_type &;
54 
55 public:
56  //! allocate empty simple vector
57  SimpleVector()
58  : size_(0), array_(nullptr)
59  { }
60 
61  //! allocate vector's memory
62  explicit SimpleVector(const size_type& sz)
63  : size_(sz), array_(nullptr) {
64  if (size_ > 0)
65  array_ = new value_type[size_];
66  }
67 
68  //! non-copyable: delete copy-constructor
69  SimpleVector(const SimpleVector&) = delete;
70  //! non-copyable: delete assignment operator
71  SimpleVector& operator = (const SimpleVector&) = delete;
72 
73  //! move-constructor
74  SimpleVector(SimpleVector&& v) noexcept
75  : size_(v.size_), array_(v.array_)
76  { v.size_ = 0, v.array_ = nullptr; }
77 
78  //! move-assignment
79  SimpleVector& operator = (SimpleVector&& v) noexcept {
80  if (&v == this) return *this;
81  delete[] array_;
82  size_ = v.size_, array_ = v.array_;
83  v.size_ = 0, v.array_ = nullptr;
84  return *this;
85  }
86 
87  //! swap vector with another one
88  void swap(SimpleVector& obj) noexcept {
89  std::swap(size_, obj.size_);
90  std::swap(array_, obj.array_);
91  }
92

96  }

146  //! resize the array to contain exactly new_size items
147  void resize(size_type new_size) {
148  if (array_) {
149  value_type* tmp = array_;
150  array_ = new value_type[new_size];
151  std::move(tmp, tmp + std::min(size_, new_size), array_);
152  delete[] tmp;
153  size_ = new_size;
154  }
155  else {
156  array_ = new value_type[new_size];
157  size_ = new_size;
158  }
159  }
160 
161  //! Zero the whole array content.
162  void fill(const value_type& v = value_type()) noexcept {
163  std::fill(array_, array_ + size_, v);
164  }
