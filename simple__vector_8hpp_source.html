34 /*!
35  * Simpler non-growing vector without initialization.
36  *
37  * SimpleVector can be used a replacement for std::vector when only a
38  * non-growing array of simple types is needed. The advantages of SimpleVector
39  * are that it does not initilize memory for POD types (-> faster), while normal
40  * structs are supported as well if default-contractible. The simple pointer
41  * types allow faster compilation and is less error prone to copying and other
42  * problems.
43  */
44 template <typename ValueType,
45  SimpleVectorMode Mode = SimpleVectorMode::Normal>
46 class SimpleVector
47 {
48 public:
49  using value_type = ValueType;
50  using size_type = size_t;
51 
52 protected:
53  //! size of allocated memory
54  size_type size_;
55 
56  //! pointer to allocated memory area
57  value_type* array_;
58 
59 public:
60  // *** simple pointer iterators
61 
62  using iterator = value_type *;
63  using const_iterator = const value_type *;
64  using reference = value_type &;
65  using const_reference = const value_type &;
66 
67 public:
68  //! allocate empty simple vector
69  SimpleVector()
70  : size_(0), array_(nullptr)
71  { }
72 
73  //! allocate vector's memory
74  explicit SimpleVector(const size_type& sz)
75  : size_(sz), array_(nullptr) {
76  if (size_ > 0)
77  array_ = create_array(size_);
78  }
79 
80  //! non-copyable: delete copy-constructor
81  SimpleVector(const SimpleVector&) = delete;
82  //! non-copyable: delete assignment operator
83  SimpleVector& operator = (const SimpleVector&) = delete;
84 
85  //! move-constructor
86  SimpleVector(SimpleVector&& v) noexcept
87  : size_(v.size_), array_(v.array_)
88  { v.size_ = 0, v.array_ = nullptr; }
89 
90  //! move-assignment
91  SimpleVector& operator = (SimpleVector&& v) noexcept {
92  if (&v == this) return *this;
93  destroy_array(array_, size_);
94  size_ = v.size_, array_ = v.array_;
95  v.size_ = 0, v.array_ = nullptr;
96  return *this;
97  }
98 
99  //! swap vector with another one
100  void swap(SimpleVector& obj) noexcept {
101  std::swap(size_, obj.size_);
102  std::swap(array_, obj.array_);
103  }
104 
105  //! delete vector
106  ~SimpleVector() {
107  destroy_array(array_, size_);
108  }

158  //! resize the array to contain exactly new_size items
159  void resize(size_type new_size) {
160  if (array_) {
161  value_type* tmp = array_;
162  array_ = create_array(new_size);
163  std::move(tmp, tmp + std::min(size_, new_size), array_);
164  destroy_array(tmp, size_);
165  size_ = new_size;
166  }
167  else {
168  array_ = create_array(new_size);
169  size_ = new_size;
170  }
171  }
172 
173  //! Zero the whole array content.
174  void fill(const value_type& v = value_type()) noexcept {
175  std::fill(array_, array_ + size_, v);
176  }
177 
178 private:
179  static ValueType * create_array(size_t size) {
180  switch (Mode)
181  {
182  case SimpleVectorMode::Normal:
183  // with normal object construction
184  return new value_type[size];
185  case SimpleVectorMode::NoInitButDestroy:
186  case SimpleVectorMode::NoInitNoDestroy:
187  // operator new allocates bytes
188  return static_cast<ValueType*>(
189  operator new (size * sizeof(ValueType)));
190  }
191  }
192 
193  static void destroy_array(ValueType* array, size_t size) {
194  switch (Mode)
195  {
196  case SimpleVectorMode::Normal:
197  // with normal object destruction
198  delete[] array;
199  return;
200  case SimpleVectorMode::NoInitButDestroy:
201  for (size_t i = 0; i < size; ++i)
202  array[i].~ValueType();
203  // intentionally fall through to next case
204  case SimpleVectorMode::NoInitNoDestroy:
205  // only deallocate memory
206  operator delete (array);
207  return;
208  }
209  }
210 };
