29 /*!
30  * This is an expected O(1) LRU cache which contains a set of key-only
31  * elements. Elements can be put() into LRU cache, and tested for existence
32  * using exists(). Insertion and touch() will remark the elements as most
33  * recently used, pushing all other back in priority. The LRU cache itself does
34  * not limit the number of items, because it has no eviction mechanism. Instead,
35  * the user program must check size() after an insert and may extract the least
36  * recently used element.
37  */
38 template <typename Key, typename Alloc = std::allocator<Key> >
39 class LruCacheSet
40 {
41 protected:
42  using List = typename std::list<Key, Alloc>;
43  using ListIterator = typename List::iterator;
44 
45  using Map = typename std::unordered_map<
46  Key, ListIterator, std::hash<Key>, std::equal_to<Key>,
47  typename Alloc::template rebind<
48  std::pair<const Key, ListIterator> >::other>;
49 
50 public:
51  explicit LruCacheSet(const Alloc& alloc = Alloc())
52  : list_(alloc), map_(alloc) { }
53 
54  //! clear LRU
55  void clear() {
56  list_.clear();
57  map_.clear();
58  }
59 
60  //! put or replace/touch item in LRU cache
61  void put(const Key& key) {
62  // first try to find an existing key
63  typename Map::iterator it = map_.find(key);
64  if (it != map_.end()) {
65  list_.erase(it->second);
66  map_.erase(it);
67  }
68 
69  // insert key into linked list at the front (most recently used)
70  list_.push_front(key);
71  // store iterator to linked list entry in map
72  map_.insert(std::make_pair(key, list_.begin()));
73  }

76  void touch(const Key& key) {
77  typename Map::iterator it = map_.find(key);
78  if (it == map_.end()) {
79  throw std::range_error("There is no such key in cache");
80  }
81  else {
82 
83  list_.splice(list_.begin(), list_, it->second);
84  }
85  }

148 /*!
149  * This is an expected O(1) LRU cache which contains a map of (key -> value)
150  * elements. Elements can be put() by key into LRU cache, and later retrieved
151  * with get() using the same key. Insertion and retrieval will remark the
152  * elements as most recently used, pushing all other back in priority. The LRU
153  * cache itself does not limit the number of items, because it has no eviction
154  * mechanism. Instead, the user program must check size() before or after an
155  * insert and may extract the least recently used element.
156  */
